anti_hungarian_russia_aid = {
	is_shown = {
		exists = c:AUS
		c:AUS = ROOT
		c:HUN = {
			has_variable = hun_fighting_war_var
			is_at_war = yes
		}
		NOT = { has_variable = asked_russian_aid_var }
	}
	
	possible = {
		any_scope_war = {
		    has_war_support = {
		        target = ROOT
		        value < 50
		    }
		}
	}
	
	when_taken = {
		custom_tooltip = anti_hungarian_russia_aid.t1
		set_variable = asked_russian_aid_var
		c:RUS = {
			trigger_event = {
				id = dual_monarchy.5
			}
		}
		every_scope_war = {
			limit = {
				has_war_goal = reduce_autonomy
			}
			add_war_war_support = {
				target = ROOT
				value = 50
			}
		}
	}

	ai_chance = {
		base = 100
	}
}

reform_ah = {
	is_shown = {
		exists = c:AUS
		c:AUS = ROOT
		has_variable = formed_ah
		NOT = { country_has_primary_culture = cu:hungarian }
	}
	
	when_taken = {
		custom_tooltip = anti_hungarian_russia_aid.t1
		add_primary_culture = cu:hungarian
	}

	ai_chance = {
		base = 100
	}
}
