dnf_events.9 = {
	type = country_event
	event_image = {
		video = "gfx/event_pictures/europenorthamerica_springtime_of_nations.bk2"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/springtime_of_nation"
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	title = dnf_events.9.t
	desc = dnf_events.9.d
	flavor = dnf_events.9.f
	duration = 3

	trigger = {}

	option = {
	name = dnf_events.9.a
	
		custom_tooltip = {
			text = dnf_release
			create_country = {
				tag = DNF
				origin = ROOT
				state = s:STATE_CENTRAL_HUNGARY.region_state:GER
				on_created = {
					c:GER = {
						every_scope_state = {
							limit = {
								OR = { 
									state_region = { is_homeland = cu:hungarian }
									state_region = { is_homeland = cu:croat }
									state_region = { is_homeland = cu:romanian }
									state_region = { is_homeland = cu:czech }
									state_region = { is_homeland = cu:slovene }
									state_region = { is_homeland = cu:slovak }
								}
							}
							set_state_owner = c:DNF
						}
						s:STATE_EAST_GALICIA = {
							every_scope_state = {
								set_state_owner = c:DNF
							}
						}
						s:STATE_WEST_GALICIA = {
							every_scope_state = {
								set_state_owner = c:DNF
							}
						}
					}
					c:DNF = {
						ig:ig_intelligentsia = {
							add_ruling_interest_group = yes
						}
					}
				add_primary_culture = cu:czech
				add_primary_culture = cu:romanian
				add_primary_culture = cu:croat
					
				}
			}
			c:GER = {
				if = {
					limit = {
						has_law = law_type:law_monarchy
					}
					create_diplomatic_pact = {
						country = c:DNF
						type = personal_union
					}
				}
				else = {
					create_diplomatic_pact = {
						country = c:DNF
						type = puppet
					}
				}
			}
		}
	}

	option = {
		name = dnf_events.9.b
		default_option = yes
		add_radicals = {
			value = 0.3
			culture = cu:romanian
		}
		add_radicals = {
			value = 0.3
			culture = cu:czech
		}
		add_radicals = {
			value = 0.3
			culture = cu:hungarian

		}
		add_radicals = {
			value = 0.3
			culture = cu:slovak
		}
		add_radicals = {
			value = 0.3
			culture = cu:croat
		}
		add_radicals = {
			value = 0.3
			culture = cu:slovene
		}
		
	}
}
