inflation.9 = { # An event to recalculate inflation and post the notification if the game rule is enabled
	type = country_event
    placement = ROOT
	hidden = yes

    trigger = {
        OR = {
			quarterly_pulse = yes
			NOT = {
				has_global_variable = first_inflation_report
			}
		}
		NOT = {
			is_country_type = decentralized
		}
    }
	
	immediate = {
		recalculate_inflation = yes
		if = {
			limit = {
				NOT = {
					has_global_variable = first_inflation_report
				}
			}
			set_global_variable = first_inflation_report
		}
		if = {
			limit = {
				has_game_rule = inflation_reports_enabled
			}
			post_notification = tgr_inflation_report
		}
	}
}
