check_if_root_is_generic_gdp = {
	scope = country

	is_shown = {
		NOT = {
			OR = {
				AND = {
					exists = c:USA
					OR = {
						this = c:USA
						is_subject_of = c:USA
					}
				}
				AND = {
					exists = c:AUS
					OR = {
						this = c:AUS
						is_subject_of = c:AUS
					}
				}
				AND = {
					exists = c:FRA
					OR = {
						this = c:FRA
						is_subject_of = c:FRA
					}
				}
				AND = {
					exists = c:RUS
					OR = {
						this = c:RUS
						is_subject_of = c:RUS
					}
				}
			}
		}
	}
}

			}
			AND = {
				exists = c:RUS
				this = c:RUS
			}
			AND = {
				exists = c:AUS
				this = c:AUS
			}
			AND = {
				exists = c:FRA
				this = c:FRA
			}
			AND = {
				exists = c:ITA
				this = c:ITA
			}
			AND = {
				exists = c:TUR
				this = c:TUR
			}
			AND = {
				exists = c:JAP

			}
			AND = {
				exists = c:USA
				this = c:USA
			}
			AND = {
				exists = c:GBR

			}
        }
    }
